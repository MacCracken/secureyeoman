Get Started in Minutes

Deploy your personal AI Yeoman with just a few commands.

terminal 
# Clone the repository
$ git clone https://github.com/MacCracken/secureyeoman.git
$ cd secureyeoman

# Install dependencies
$ npm install

# Configure environment
$ cp .env.example .env
# Edit .env with your API keys

# Start SecureYeoman
$ npm run dev
terminal 
# Start with Docker Compose
$ docker compose up -d

# With MCP service
$ docker compose --profile mcp up -d

# Fresh start
$ docker compose down -v
$ docker compose up -d
terminal 
# Deploy with Helm (EKS, GKE, AKS)
$ helm install friday deploy/helm/friday \
    --namespace friday --create-namespace \
    --set secrets.postgresPassword=your-password \
    --set database.host=your-db.example.com

# Production deployment
$ helm install friday deploy/helm/friday \
    -f deploy/helm/friday/values-production.yaml \
    --namespace friday-production --create-namespace

# Verify
$ helm test friday --namespace friday
terminal 
# Prerequisites
- Node.js 20 LTS+
- npm or pnpm
- AI Provider API Key

# Security Keys (32+ chars each)
SECUREYEOMAN_SIGNING_KEY="..."

SECUREYEOMAN_ENCRYPTION_KEY="..."
